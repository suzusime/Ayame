<スクリプト> ::= <行>
               | <行> "改行" <スクリプト>

<行> ::= "タブ" <式> //<式>が台詞。発話者名は承前。
       | <式> "タブ" <式> //1つめの<式>が発話者名、2つめの<式>が台詞
       | "#" <普通文字列> //ラベル

<式> ::= <文字列>
       | "[" <関数呼出> "]"
       | "区切り"
       | <文字列> <式>
       | "[" <関数呼出> "]" <式>
       | "区切り" <式>

<文字列> ::= "普通文字列"
           | "変数"

<関数呼出> ::= <文字列>
             | "[" <関数呼出> "]"
             | <文字列> "区切り" <関数呼出>
             | "[" <関数呼出> "]" "区切り" <関数呼出>
